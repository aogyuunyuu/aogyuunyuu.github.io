テキストランダム抽出器


import random def extract_random_lines(event): input_text = Element("inputText").value lines = input_text.splitlines() unique_lines = list(set(lines)) try: line_count = int(Element("lineCount").value) except ValueError: line_count = 10 # デフォルトは10行 line_count = min(line_count, len(unique_lines)) random.shuffle(unique_lines) output_lines = unique_lines[:line_count] Element("outputText").element.value = "\n".join(output_lines) Element("extractButton").element.onclick = extract_random_lines
